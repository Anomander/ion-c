00015 /*
00016  use cases:
00017  
00018  read only stream - (aka (can write == false)
00019   --next char
00020       if not end of buffer - done
00021       if user buffer - fail
00022       find next page
00023       if not found
00024         if mark allocate new page
00025         fill page from last read position
00026   --unread
00027       if beginning of page
00028         find previous page
00029         else (not found)
00030           create temp page
00031           mark as fake
00032         save current page
00033         make new (found or temp) page current @ end of page
00034       back up in current page
00035   --write - fail
00036   --seek - fail
00037   --skip N
00038       next char N times
00039  
00040   write only stream - (aka (can read == false)
00041     --next char - fail 
00042     --unread - fail
00043     --write
00044       if no room
00045         write current page to disk
00046         release current page
00047       get new page
00048       fill page
00049       make page current @ beginning of buf
00050       update dirty pointers
00051     --seek - fail
00052     --skip - fail
00053     
00054 ----- read only and write only streams are not seekable
00055 
00056   user buffer
00057     --to initialize allocate stream and 1 page set up page to point to user buffer
00058     --next char
00059       if end of buffer fail with EOF
00060     --unread
00061       if beginning of buffer - fail
00062       back up in current buffer
00063     --write
00064       if write past buffer length - fail
00065       write at current position
00066       update dirty pointers
00067     --seek
00068       if position outside buffer - fail
00069       set position
00070     --skip
00071       compute new position
00072       seek to new position
00073 
00074   seekable (read/write)
00075     --next char - fail
00076     --unread - fail
00077     --write
00078       if no room in current
00079         find next page
00080         if not found
00081           make page
00082           seek to page
00083           fill page
00084       write data at page
00085       update dirty pointers
00086     --seek
00087       find page
00088       if not found
00089         make page
00090         seek in file to page start
00091         fill page
00092       make page current
00093       set position @ position in page
00094     --skip
00095       compute new position
00096       seek to new position
00097 
00098   ----make page
00099         when not buffering
00100           flush current page
00101           release current page
00102         allocate page
00103 
00104   ----make page current
00105         if page != current page
00106           flush current page
00107           if not buffering release current page
00108         set up stream members to new page
00109         clear dirty pointers
00110  *
00111  */

00171 struct _ion_stream
00172 {
00173   ION_STREAM_FLAG   _flags;        // CAN_READ, CAN_WRITE, RANDOM_ACCESS, CONSOLE_HANDLING, 
00174                                   // IS_FILE_BACKED, BUFFER_ALL, IS_USER_BUFFER, 
00175                                   // IS_OPEN, IS_DIRTY, IS_AT_EOF, IS_FAKE_PAGE, 
00176   FILE            *_fp;           // current file for read or read/write (will be NULL if user buffer)
00177   BYTE            *_buffer;       // current page of data in use
00178   SIZE             _buffer_size;  // current buffer size, this is page size if the is a paged stream, otherwise it's the one and only buffer length
00179   POSITION         _offset;       // offset of first byte of buffer in source
00180   BYTE            *_curr;         // position = _offset + (_curr - _buffer)
00181   BYTE            *_limit;        // end of buffered data
00182 
00183   POSITION         _mark;         // -1 for no mark otherwise the file position where the mark started
00184 
00185   BYTE            *_dirty_start;  // pointer to first dirty byte in current buffer
00186   SIZE             _dirty_length; // number of dirty bytes (only contiguous bytes in the current buffer are allowed to be dirty)
00187 };
