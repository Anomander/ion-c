00015 /*
00016  * updated versions of the file stream handling support for
00017  * the IonC readers and writers.
00018  * 
00019  *  fn_fill -  updates the streams buffer, curr, limit and max
00020  *             fields.  Should return -1 on eof or bytes read
00021  *             or error.  For a writable stream this may need to
00022  *             append an empty buffer.
00023  * 
00024  *  fn_unread- unreads a byte when there is no room in the current
00025  *             buffer to hold the unread byte. This will not get
00026  *             called with an EOF value. Normal behavior is to create 
00027  *             an empty buffer that "preceeds" the current buffer and
00028  *             write the byte there.  It may use the the actual previous
00029  *             buffer.  Or it may fail.
00030  * 
00031  *  fn_flush - writes any dirty data out to the output stream.
00032  * 
00033  *  fn_close - releases any resources as needed.
00034  *
00035  
00036  
00037  MISSING: skip, fixes for block read/unread, collapse some functions
00038  
00039  
00040  use cases:
00041  
00042  read only stream - (aka (can write == false)
00043   --next char
00044       if not end of buffer - done
00045       if user buffer - fail
00046       find next page
00047       if not found
00048         if mark allocate new page
00049         fill page from last read position
00050   --unread
00051       if beginning of page
00052         find previous page
00053         else (not found)
00054           create temp page
00055           mark as fake
00056         save current page
00057         make new (found or temp) page current @ end of page
00058       back up in current page
00059   --write - fail
00060   --seek - fail
00061   --skip N
00062       next char N times
00063  
00064   write only stream - (aka (can read == false)
00065     --next char - fail 
00066     --unread - fail
00067     --write
00068       if no room
00069         write current page to disk
00070         release current page
00071       get new page
00072       fill page
00073       make page current @ beginning of buf
00074       update dirty pointers
00075     --seek - fail
00076     --skip - fail
00077     
00078 ----- read only and write only streams are not seekable
00079 
00080   user buffer
00081     --to initialize allocate stream and 1 page set up page to point to user buffer
00082     --next char
00083       if end of buffer fail with EOF
00084     --unread
00085       if beginning of buffer - fail
00086       back up in current buffer
00087     --write
00088       if write past buffer length - fail
00089       write at current position
00090       update dirty pointers
00091     --seek
00092       if position outside buffer - fail
00093       set position
00094     --skip
00095       compute new position
00096       seek to new position
00097 
00098   seekable (read/write)
00099     --next char - fail
00100     --unread - fail
00101     --write
00102       if no room in current
00103         find next page
00104         if not found
00105           make page
00106           seek to page
00107           fill page
00108       write data at page
00109       update dirty pointers
00110     --seek
00111       find page
00112       if not found
00113         make page
00114         seek in file to page start
00115         fill page
00116       make page current
00117       set position @ position in page
00118     --skip
00119       compute new position
00120       seek to new position
00121 
00122   ----make page
00123         when not buffering
00124           flush current page
00125           release current page
00126         allocate page
00127 
00128   ----make page current
00129         if page != current page
00130           flush current page
00131           if not buffering release current page
00132         set up stream members to new page
00133         clear dirty pointers
00134  *
00135  */
