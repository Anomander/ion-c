00018 //
00019 // overall format is (from ion wiki):
00020 //
00021 //      null.timestamp                   // A null timestamp value
00022 //      
00023 //      2007-02-23                       // A day, equivalent to 2007-02-23T00:00:00-00:00
00024 //      2007-02-23T12:14Z                // Seconds are optional, but timezone is not
00025 //      2007-02-23T12:14:33.079-08:00    // A timestamp with millisecond precision and PST local time
00026 //      2007-02-23T20:14:33.079Z         // The same instant in UTC ("zero" or "zulu")
00027 //      2007-02-23T20:14:33.079+00:00    // The same instant with explicit local offset
00028 //
